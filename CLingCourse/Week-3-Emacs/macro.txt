3. $glue macro
;; Keyboard Macro Editor.  Press C-c C-c to finish; press C-x k RET to cancel.
;; Original keys: RET C-x h M-w C-x b g l u e d . t x t RET C-y RET C-x b 7 RET <down>

Command: last-kbd-macro
Key: none

Macro:

RET			;; dired-find-file
C-x h			;; mark-whole-buffer
M-w			;; kill-ring-save
C-x b			;; switch-to-buffer
g			;; revert-buffer
l			;; dired-do-redisplay
u			;; dired-unmark
e			;; dired-find-file
d			;; dired-flag-file-deletion
.			;; dired-clean-directory
t			;; dired-toggle-marks
x			;; dired-do-flagged-delete
t			;; dired-toggle-marks
RET			;; dired-find-file
C-y			;; yank
RET			;; dired-find-file
C-x b			;; switch-to-buffer
7			;; digit-argument
RET			;; dired-find-file
<down>			;; next-line
